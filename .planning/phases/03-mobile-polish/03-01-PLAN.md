```js
// Cordy's Crafts — useBreakpoint hook
// Returns 'mobile' | 'tablet' | 'desktop' based on window.innerWidth.
// Breakpoints (D-02): ≤640px → mobile, 641-1024px → tablet, >1024px → desktop.
// Updates on resize; cleans up listener on unmount.

import { useState, useEffect } from 'react'

export function useBreakpoint() {
  const getBreakpoint = () => {
    const w = window.innerWidth
    if (w <= 640) return 'mobile'
    if (w <= 1024) return 'tablet'
    return 'desktop'
  }

  const [bp, setBp] = useState(getBreakpoint)

  useEffect(() => {
    const onResize = () => setBp(getBreakpoint())
    window.addEventListener('resize', onResize)
    return () => window.removeEventListener('resize', onResize)
  }, [])

  return bp
}
```

No debounce — resize events at this scale are cheap (per D-01 discretion note). No default export — named export only, consistent with `useProducts` and `useClasses`.
